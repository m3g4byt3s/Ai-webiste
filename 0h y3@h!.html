  <script>

    // Tiny dataset to mix-and-match

    const EMOJIS = ["🌿","⚡","🚀","✨","🧩","🎲","🎧","🕹️","🌈","🔥","💡","📦","🛸","🐙","🍥"];

    const ADJECTIVES = ["curious","wonky","brave","sleepy","glowing","timid","cheeky","retro","pixelated","whispering"];

    const NOUNS = ["algorithm", "star", "notebook", "dragon", "socket", "cable", "mug", "server", "daemon", "constellation"];

    const QUOTES = [

      "Sometimes the glitch is the point.",

      "Write little things, ship often.",

      "Pixels have feelings too.",

      "Refactor the mess; keep the magic.",

      "If it compiles, it's an adventure.",

      "Bugs are future features in disguise."

    ];

    function rand(arr){ return arr[Math.floor(Math.random()*arr.length)]; }

    function rnd(min,max){ return Math.floor(Math.random()*(max-min+1))+min; }

    function hsl(h,s,l){ return `hsl(${h} ${s}% ${l}%)`; }

    function makeCard(){

      // random base hue and derived colors

      const hue = rnd(0,359);

      const bgL = rnd(10,18);

      const accentL = rnd(50,70);

      const accentS = rnd(55,90);

      const emoji = rand(EMOJIS);

      const title = `${rand(ADJECTIVES)} ${rand(NOUNS)}`;

      const quote = rand(QUOTES);

      const id = Math.random().toString(36).slice(2,8);

      const card = document.createElement('article');

      card.className = 'card';

      // inline style for subtle accent

      card.style.background = `linear-gradient(135deg, ${hsl(hue,20, bgL)} 0%, ${hsl(hue+20,10, (bgL-2))} 100%)`;

      card.style.borderColor = hsl(hue, 20, bgL+6);

      card.innerHTML = `

        <div class="badge" style="background:${hsl(hue,accentS,accentL)}33;border:1px solid ${hsl(hue,accentS,accentL)}22">${title}</div>

        <div class="emoji">${emoji}</div>

        <div class="quote">${quote}</div>

        <div class="meta">

          <div>id: ${id}</div>

          <div>${rnd(1,99)}% random</div>

        </div>

        <div class="pattern">${emoji}</div>

      `;

      return card;

    }

    function renderGrid(count=8){

      const grid = document.getElementById('grid');

      grid.innerHTML = ''; // reset

      for(let i=0;i<count;i++){

        grid.appendChild(makeCard());

      }

    }

    // download current HTML state as a standalone file

    function downloadHTML(){

      const blob = new Blob([document.documentElement.outerHTML], {type:'text/html'});

      const url = URL.createObjectURL(blob);

      const a = document.createElement('a');

      a.href = url; a.download = 'random_playground.html';

      document.body.appendChild(a); a.click(); a.remove();

      URL.revokeObjectURL(url);

    }

    // init

    document.getElementById('shuffleBtn').addEventListener('click', ()=> renderGrid(rnd(6,12)));

    document.getElementById('downloadBtn').addEventListener('click', downloadHTML);

    // render first time

    renderGrid(8);

    // small keyboard shortcut: press R to reshuffle

    window.addEventListener('keydown', (e)=>{

      if(e.key.toLowerCase() === 'r' && !e.metaKey && !e.ctrlKey){

        renderGrid(rnd(6,12));

      }

    });
